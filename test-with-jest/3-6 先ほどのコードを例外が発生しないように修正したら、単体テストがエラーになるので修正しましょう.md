### コード追加

配列の長さが 0 の場合、0 を返す処理を追加

```tsx
// functions.ts
export const sumOfArray = (numbers: number[]): number => {
  // 配列が空の場合は0を返す
  if (numbers.length === 0) {
    return 0;
  }
  return numbers.reduce((a: number, b: number): number => a + b);
};
```
